Table users {
  id integer [primary key]
  login varchar
  password char[60] [note: "bcrypt hash"]
}

Table lists {
  id integer [primary key]
  owner_id integer
  name varchar
  updated_at timestamp
  image_path varchar
  share_code varchar
}

Table items {
  id integer [primary key]
  list_id integer
  name varchar
  description varchar
  count integer
  updated_at timestamp
  deleted boolean
}

Table shared_with {
  id integer [primary key]
  user_id integer
  list_id integer
}

Ref: items.list_id > lists.id
Ref: users.id > lists.owner_id
Ref: shared_with.list_id > lists.id
Ref: shared_with.user_id > users.id
